&НаСервере
Процедура ПриСозданииНаСервере(Отказ, СтандартнаяОбработка)
	Объект.Сводные = Истина;
	Объект.СУчетомОплаченных = Истина;
	Объект.ДатаВыгрузки = ТекущаяДата();
	Организация = Справочники.Организации.ОрганизацияПоУмолчанию();
	
	ТЗОснованияВыплат = РеквизитФормыВЗначение("ОснованияВыплат");
	ТЗОснованияВыплат = Документы.ВедомостьНаВыплатуЗарплатыВБанк.СоответствиеОснованийВыплат();  
	ЗначениеВРеквизитФормы(ТЗОснованияВыплат, "ОснованияВыплат");
	
	ВыводитьГоловноеПодразделение = ПолучитьФункциональнуюОпцию("ВыводитьГоловноеПодразделение");
	Элементы.ГоловноеПодразделение.Видимость = ВыводитьГоловноеПодразделение;
	
	Если ВыводитьГоловноеПодразделение Тогда
		НастройкиПользователя = ЗарплатаКадры.ПолучитьЗначенияНастроекПользователя(Пользователи.АвторизованныйПользователь());	
		Если Параметры.Свойство("ГоловноеПодразделение") И ЗначениеЗаполнено(Параметры.ГоловноеПодразделение) Тогда
			Объект.ГоловноеПодразделение = Параметры.ГоловноеПодразделение;
		Иначе
			Если ЗначениеЗаполнено(НастройкиПользователя.ОсновноеПроизводственноеУправление) Тогда
				Объект.ГоловноеПодразделение = НастройкиПользователя.ОсновноеПроизводственноеУправление;
			Иначе
				Объект.ГоловноеПодразделение = НастройкиПользователя.ОграничиватьОтборСотрудниковПоПодразделению;
			КонецЕсли;
		КонецЕсли;	
	КонецЕсли;
	
	// СтандартныеПодсистемы.Печать
	УправлениеПечатью.ПриСозданииНаСервере(ЭтаФорма, Элементы.ВедомостиКоманднаяПанель);
	УправлениеПечатью.ПриСозданииНаСервере(ЭтаФорма, Элементы.ВедомостиСводныеКоманднаяПанель);

	// Конец СтандартныеПодсистемы.Печать

&НаКлиенте
Процедура ПересчитатьОбщуюСумму()
	
	ОбщаяСумма = 0;
	
	Для Каждого Стр Из Объект.Проекты Цикл
		Стр.Сумма = 0;
	Конеццикла;
	
	МаксПериоды = СоответствиеМаксПериодовРегистрации();
	
	Для Каждого Строка Из Объект.Ведомости Цикл
		Если Строка.Выгрузить Тогда
			ОбщаяСумма = ОбщаяСумма + Строка.Сумма;
			МассивОснований = ОснованияВыплат.НайтиСтроки(Новый Структура("СпособВыплаты", Строка.СпособВыплаты));
			Отбор = Новый структура("ЗарплатныйПроект",Строка.ЗарплатныйПроект);
			РеквизитыЗП = ПолучитьРеквизитыЗП(Строка.ЗарплатныйПроект);
			Массив = Объект.Проекты.НайтиСтроки(Отбор);
			Если Массив.Количество() = 0 Тогда
				НС = Объект.Проекты.Добавить();
				НС.ЗарплатныйПроект = Строка.ЗарплатныйПроект;
			Иначе
				НС = Массив[0];
			КонецЕсли;
			НС.НомерВедомости = Макс(НС.НомерВедомости,Строка.НомерВедомости);
			НС.Сумма = НС.Сумма + Строка.Сумма;
			НС.МесяцНачисления = МаксПериоды.Получить(Строка.ЗарплатныйПроект);
			НС.Основание = МассивОснований[0].Основание;
			НС.НомерПлатежногоПоручения = Строка.НомерПлатежногоПоручения;
			
			Если ЗначениеЗаполнено(РеквизитыЗП.Комиссия) Тогда
				НС.Комиссия = (НС.Сумма*РеквизитыЗП.Комиссия)/100;
			КонецЕсли;
		КонецЕсли;	
	КонецЦикла;	
	
	Отбор = Новый структура("Сумма",0);
	Массив = Объект.Проекты.НайтиСтроки(Отбор);
	
	Для Каждого СтрМ Из Массив Цикл
		Объект.Проекты.Удалить(СтрМ);
	КонецЦикла;
	
	Объект.ОбщаяСумма = ОбщаяСумма;
	
	УстановитьНомераВедомостейНаСервере();

&НаСервере
Процедура УстановитьОтметкуОплаченоВЗагруженныхДокументах()
	
	ТЗ = Объект.Ведомости.Выгрузить();
	ТЗ.Свернуть("Выгрузить,Ведомость");
	
	Для Каждого Стр Из Тз Цикл
		Если Стр.Выгрузить Тогда
			НомВедомостиМакс = 0;
			ОплаченныйДок = Стр.Ведомость.ПолучитьОбъект();
			Для Каждого СтрЗП Из ОплаченныйДок.Зарплата Цикл
				Отбор = Новый Структура("ЗарплатныйПроект",СтрЗП.ЗарплатныйПроект);
				Массив = Объект.Проекты.НайтиСтроки(Отбор);
				Для Каждого СтрМ Из Массив Цикл
					СтрЗП.НомерВедомости = СтрМ.НомерВедомости;
					НомерВедомостиМакс = Макс(СтрЗП.НомерВедомости,НомВедомостиМакс);
				КонецЦикла;
			КонецЦикла;
			ОплаченныйДок.Оплачено = Истина;
			ОплаченныйДок.НомерВедомости = НомерВедомостиМакс;
			Попытка
				ОплаченныйДок.Записать(РежимЗаписиДокумента.Проведение);
			Исключение
			КонецПопытки;				
		КонецЕсли;
	КонецЦикла;

Процедура ПересчитатьОбщуюСуммуСводная()
	
	ОбщаяСумма = 0;
	
	Для Каждого Стр Из Объект.Проекты Цикл
		Стр.Сумма = 0;
	Конеццикла;
	
	МаксПериоды = СоответствиеМаксПериодовРегистрации();
	
	ТЗВед1 =	ПолучитьТаблицуВедомостей() ;

	Для Каждого СтрокаСв Из Объект.ВедомостиСводные Цикл
		Если СтрокаСв.Выгрузить Тогда
			
			
		
			
		    	ОбщаяСумма = ОбщаяСумма + СтрокаСв.Сумма;
	    		Отбор = Новый структура("ЗарплатныйПроект",СтрокаСв.ЗарплатныйПроект);
				РеквизитыЗП = ПолучитьРеквизитыЗП(СтрокаСв.ЗарплатныйПроект);
				МассивВедомости = Объект.Ведомости.НайтиСтроки(Отбор);

		//	Для Каждого Строка Из МассивВедомости Цикл
				
				МассивОснований = ОснованияВыплат.НайтиСтроки(Новый Структура("СпособВыплаты", СтрокаСв.СпособВыплаты));
		//		РеквизитыЗП = ПолучитьРеквизитыЗП(СтрокаСв.ЗарплатныйПроект);
				Массив = Объект.Проекты.НайтиСтроки(Отбор);
				Если Массив.Количество() = 0 Тогда
					НС = Объект.Проекты.Добавить();
					НС.ЗарплатныйПроект = СтрокаСв.ЗарплатныйПроект;
				Иначе
					НС = Массив[0];
				КонецЕсли;
				НС.НомерВедомости = СтрокаСв.НомерВедомости;//Макс(НС.НомерВедомости,Строка.НомерВедомости);
				НС.Сумма = НС.Сумма + СтрокаСв.Сумма;//Строка.Сумма;
				НС.МесяцНачисления = МаксПериоды.Получить(СтрокаСв.ЗарплатныйПроект);
				НС.Основание = МассивОснований[0].Основание;
				НС.НомерПлатежногоПоручения = СтрокаСв.НомерПлатежногоПоручения;
				
				Если ЗначениеЗаполнено(РеквизитыЗП.Комиссия) Тогда
					НС.Комиссия = (НС.Сумма*РеквизитыЗП.Комиссия)/100;
				КонецЕсли;
				
			//КонецЦикла;			
		КонецЕсли;	
	КонецЦикла;	
	
	Отбор = Новый структура("Сумма",0);
	Массив = Объект.Проекты.НайтиСтроки(Отбор);
	
	Для Каждого СтрМ Из Массив Цикл
